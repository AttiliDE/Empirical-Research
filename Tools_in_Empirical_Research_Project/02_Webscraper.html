url = 'https://es.wikipedia.org/wiki/Anexo:Municipios_de_Espaa_por_poblacin'
response = requests.get(url)

# pase the website to a beautifulsoup object
soup = BeautifulSoup(response.content, 'html.parser')

# Find all the tables:
tables = soup.find_all("table", class_="wikitable")

# create a pandas dataframe to store the values
columns = ['ID', 'Municipality', 'Population', 'Province', 'Region']
df_cities = pd.DataFrame(columns = columns)

data = []

for table in tables:
    rows = table.find_all('tr')
    # Loop over the rows and extract the data for each municipality
    for row in rows[1:]:  # Start from index 1 to skip the header row
        # Find all cells in the row
        cells = row.find_all("td")

        # Extract the data from each cell
        position = cells[0].text.strip()
        municipality = cells[1].text.strip()
        population = int(cells[2].text.strip().replace(' ', ''))
        # code is given us the name of the municipality twice. We only need one of them
        string_len = len(cells[3].text.strip())
        province = cells[3].text.strip()[0:int(string_len/2)]
        try:
            string_len = len(cells[4].text.strip())
            region = cells[4].text.strip()[0:int(string_len/2)]
            
        except IndexError:
            region = province
        
        data.append([position, municipality, population, province, region])
        
df = pd.DataFrame(data = data, columns = columns)
df.set_index('ID', inplace = True)

# Import csv to be used in R.
df.to_csv('../02_RCodes/01_SpainCities.csv')

from bs4 import BeautifulSoup
import requests

column_entries = []
failed_queries = []
counter = 0
for number in unique_numbers:
    time.sleep(2)
    url = f'https://www.ine.es/jaxiT3/Datos.htm?t={number}'
    initial_length = len(queries)
    while len(queries) == initial_length:
        response = requests.get(url)

        # parse the url to beautifulsoup
        soup = BeautifulSoup(response.text, 'html.parser')
        try:
            # obtain the name of the province
            province_name = soup.find('span', class_="cab2").text

            # get the tables and the rows returned by the html
            rows = soup.find_all('table')[0].find_all('tr')

            for row in rows:
                columns = row.find_all(['th', 'td'])
                if len(columns) > 1:
                    column_entries.append([columns[0].text, columns[1].text, province_name, number])
                else:
                    column_entries.append([columns[0].text, '', '', ''])

            #every time a query is successful, we wil remove it from the the queries list.
            queries.remove(number)
            counter += 1
            print('success', counter)
        except:
            failed_queries.append(number)

We are interested specifically in the section 'Ask&offset=0&limit=20' of the link above. We will set the limit to be displayed to be equal to 500. Each offset will display another 500 cities. In total the offset goes from 0 to 5000. If we query an offset above 5000, the query will return the first page again. We actually would like to obtain the data on the column superficie, which we will use to calculate the density per city. We will not use the Density (Densidad) stated in this table, because the information we have about the current population is newer than the one stated in this table (2019 vs. 2022). Population movements is common but surface is in general a stationary measure.
